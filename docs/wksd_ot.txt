WK(B) Script Documentation
Orders/tasks
------------------------------------------------------------------------------

ORDER_ASSIGNMENT "OrderAssignment"
	ORDER_TO_ASSIGN "Order"
	ORDER_ASSIGNMENT_MODE DO_FIRST/DO_LAST/FORGET_EVERYTHING_ELSE
// optional:
	ORDER_TARGET <FINDER>
	ASSIGN_AS_HIGH_PRIORITY
END_ORDER_ASSIGNMENT

ORDER "Order"
	CLASS_TYPE OBJECT_REFERENCE/MOVE/UPGRADE/SPAWN/BUILD/REPAIR

// Optional:
	IN_ORDER_CATEGORY "OrderCategory"
	CONDINTION_IMPOSSIBLE/CONDITION_WAIT/CONDITION_WARNING "Condition"
	FLAG CANNOT_INTERRUPT_ORDER/CYCLE_ORDER/OVERRIDE_NON_INTERRUPTIBLE_ORDERS
	ON_CLICK CANCEL_ORDER
	CANCELLATION_SEQUENCE \n <actseq> \n END_CANCELLATION_SEQUENCE
	TERMINATION_SEQUENCE \n <actseq> \n END_TERMINATION_SEQUENCE

// Can be written multiple times:
	PRE_CONDITION "Equation"
	USE_TASK "Task"
END_ORDER

TASK "Task"
	CLASS_TYPE OBJECT_REFERENCE/MOVE/FACE_TOWARDS/UPGRADE/SPAWN/REPAIR // no "BUILD" used

// Optional:
	TASK_TARGET <FINDER>
	IN_TASK_CATEGORY "TaskCategory"

	ATTACHMENT_PATH_CATEGORY ACTIVITY_PATH/ATTACK...
	IDENTIFY_TARGET_EACH_CYCLE
	PROXIMITY_REQUIREMENT <VALUE>
	USE_PREVIOUS_TASK_TARGET
	TERMINATE_ENTIRE_ORDER_IF_NO_TARGET
	REJECT_TARGET_IF_IT_IS_TERMINATED
	PLAY_ANIMATION_ONCE
	PLAY_ANIMATION_TO_FIT <VALUE>

	PLAY_ANIMATION TAG "AnimTag"
/*or*/	PLAY_ANIMATION FROM_SELECTION
		ON_EQUATION_SUCCESS "Equation" "AnimTag"
		ON_EQUATION_SUCCESS "Equation" "AnimTag"
		// ...
		DEFAULT_ANIMATION "AnimTag"
	END_PLAY_ANIMATION

	START_SEQUENCE \n <actseq> \n END_START_SEQUENCE
	CANCELLATION_SEQUENCE \n <actseq> \n END_CANCELLATION_SEQUENCE
	TERMINATION_SEQUENCE \n <actseq> \n END_TERMINATION_SEQUENCE
	TRIGGER ANIMATION_LOOP/UNINTERRUPTIBLE_ANIMATION_LOOP \n <actseq> \n END_TRIGGER
	TRIGGER COLLISION/STRUCK_FLOOR \n <actseq> \n END_TRIGGER (for MISSILEs)
	PROXIMITY_SATISFIED_SEQUENCE \n <actseq> \n END_PROXIMITY_SATISFIED_SEQUENCE
	PROXIMITY_DISSATISFIED_SEQUENCE ("Equation" <FINDER>) \n <actseq> \n END_PROXIMITY_DISSATISFIED_SEQUENCE
	RESUMPTION_SEQUENCE \n <actseq> \n END_RESUMPTION_SEQUENCE

// Can be written multiple times:
	FILTER_CONDITION "Equation"
	TRIGGER TIMER <VALUE> \n <actseq> \n END_TRIGGER
	TRIGGER ATTACHMENT_POINT "SP_" OFF_TO_ON \n <actseq> \n END_TRIGGER
END_TASK
